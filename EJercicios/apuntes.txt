	AGRUPAMIENTO:
		SELECTOR1,SELECTOR2{ATRIBUTO1:VALOR1; ATRIBUTO2:VALOR2;}

	A ADYACENES:
		SELECTOR1 + SELECTOR2(A LOS HERMANOS EN LA JERARQUIA)
